procedimento verMenu ()
var
inicio

funcao fibo(x:inteiro):inteiro
var
inicio

   se (x=0) entao
      retorne 0
   senao
      se (x=1) entao
         retorne 1
      senao
         se (x>1) entao
            retorne fibo(x-1)+(x-2)
         fimse
      fimse
   fimse

fimfuncao

funcao fat (x:inteiro) :inteiro
var
inicio
   se (x=1) entao
      retorne 1
   senao
      retorne x*fat(x-1)
   fimse
fimfuncao

funcao somatorio (x:inteiro) :inteiro
var i:inteiro
inicio
   se (x=1) entao
      retorne 1
   senao
      retorne x+somatorio(x-1)
   fimse

fimfuncao

funcao somaVet (x:inteiro) :inteiro
var
inicio
   se (x=1) entao
      retorne n[1]
   senao
      retorne n[1]+somaVet(x-1)
   fimse
fimfuncao

funcao cres (x:inteiro) :inteiro
var
inicio
   se (x=0) entao
      retorne 0
      escreva(x)
   senao
      retorne cres(x-1)
      escreva(x)
   fimse
fimfuncao

funcao poten (x,y:inteiro) :inteiro
var
inicio
   se (y=1) entao
      retorne x
   senao
      se (y>=2) entao
         retorne x*poten(x,y-1)
      fimse
   fimse
fimfuncao
